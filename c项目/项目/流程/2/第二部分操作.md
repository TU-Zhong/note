```
// 确保文件顶部有 #include "core/protocol_parser.h" 和 <string.h>, <stdio.h>
#include "core/protocol_parser.h"
#include <string.h>
#include <stdio.h>

// ... build_entry_packet 等函数 ...

/*
 * 函数定义：parse_message
 * 功能：解析收到的原始网络数据包字符串，并将各部分填充到 parsed_message_t 结构体中。
 */
int parse_message(const char* packet, size_t packet_len, parsed_message_t* out_msg) {
    if (packet == NULL || out_msg == NULL || packet_len == 0) {
        return -1;
    }

    // 使用 sscanf_s 安全地解析格式化的主干部分
    // %[^:] 是一个强大的格式说明符，意为“读取所有不包含冒号的字符”
    unsigned int command_and_options;
    int items_scanned = sscanf_s(packet, "%d:%llu:%[^:]:%[^:]:%u:",
        &out_msg->version,
        &out_msg->packet_no,
        out_msg->sender_name, (unsigned)_countof(out_msg->sender_name),
        out_msg->sender_host, (unsigned)_countof(out_msg->sender_host),
        &command_and_options);

    if (items_scanned != 5) {
        return -2; // 格式不匹配，解析失败
    }
    
    // 从命令字中分离出基础命令和选项
    out_msg->command = command_and_options & 0xFF; // 低8位是基础命令
    out_msg->option = command_and_options;         // 保存完整的命令字以检查选项

    // 找到附加信息部分的起始位置
    const char* additional_start = packet;
    for (int i = 0; i < 5; ++i) {
        additional_start = strchr(additional_start, ':');
        if (additional_start == NULL) return -2; // 格式错误
        additional_start++; // 跳过冒号
    }

    // 计算并拷贝附加信息部分
    size_t main_part_len = additional_start - packet;
    if (main_part_len >= packet_len) {
        // 没有附加信息
        out_msg->additional_section[0] = '\0';
    } else {
        size_t additional_len = packet_len - main_part_len;
        // 确保不会溢出目标缓冲区
        if (additional_len >= sizeof(out_msg->additional_section)) {
            additional_len = sizeof(out_msg->additional_section) - 1;
        }
        memcpy(out_msg->additional_section, additional_start, additional_len);
        out_msg->additional_section[additional_len] = '\0'; // 确保字符串结尾
    }

    return 0; // 解析成功
}
```

```
/*
 * 函数定义：communicator_listen_loop (线程主函数)
 * 功能：在后台循环监听网络，接收UDP包，并将其分发给处理器。
 */
static DWORD WINAPI communicator_listen_loop(LPVOID param) {
    communicator_t* comm = (communicator_t*)param;
    char recv_buffer[MAX_PACKET_LEN];
    struct sockaddr_in sender_addr;
    int sender_addr_len = sizeof(sender_addr);

    printf("Listener thread started.\n");

    while (comm->is_running) {
        // 1. 调用 recvfrom 等待接收数据，这是一个阻塞操作
        int bytes_received = recvfrom(comm->sock, recv_buffer, sizeof(recv_buffer) - 1, 0,
                                      (struct sockaddr*)&sender_addr, &sender_addr_len);

        if (bytes_received == SOCKET_ERROR) {
            // 如果 is_running 为 false，说明是主动关闭，正常退出
            if (!comm->is_running) {
                break;
            }
            printf("Error: recvfrom failed with error: %d\n", WSAGetLastError());
            continue;
        }
        
        // 确保接收的数据是有效的C字符串
        recv_buffer[bytes_received] = '\0';

        // 2. 解析数据包
        parsed_message_t parsed_msg = {0};
        if (parse_message(recv_buffer, bytes_received, &parsed_msg) == 0) {
            // 获取发送方IP地址
            const char* source_ip = inet_ntoa(sender_addr.sin_addr);
            uint16_t source_port = ntohs(sender_addr.sin_port);

            // 3. 将解析后的消息交给处理器
            process_packet(comm, source_ip, source_port, &parsed_msg);
        }
    }

    printf("Listener thread exiting.\n");
    return 0;
}


/*
 * 函数定义：process_packet (内部辅助函数)
 * 功能：根据消息的命令号，执行相应的逻辑。
 */
static void process_packet(communicator_t* comm, const char* source_ip, uint16_t source_port, const parsed_message_t* msg) {
    // 忽略自己发给自己的广播包
    if (strcmp(source_ip, "127.0.0.1") == 0 || strcmp(source_ip, comm->host_name) == 0) {
         // 注意：此IP判断不完全可靠，但用于调试阶段足够
         // 更好的方法是比较包编号或生成一个唯一实例ID
    }
    
    printf("Received command %u from %s:%u\n", msg->command, source_ip, source_port);
    
    user_node_t new_user = {0};
    strcpy_s(new_user.userName, sizeof(new_user.userName), msg->sender_name);
    strcpy_s(new_user.hostName, sizeof(new_user.hostName), msg->sender_host);
    strcpy_s(new_user.ipAddress, sizeof(new_user.ipAddress), source_ip);
    new_user.port = source_port;

    switch (msg->command) {
        case IPMSG_BR_ENTRY:
            printf("User '%s' entered.\n", msg->sender_name);
            user_list_add(comm->user_list, &new_user);
            // 响应对方，告诉他我也在线
            communicator_send_ans_entry(comm, source_ip, source_port);
            break;

        case IPMSG_ANSENTRY:
            printf("Received answer from '%s'.\n", msg->sender_name);
            user_list_add(comm->user_list, &new_user);
            break;
            
        case IPMSG_BR_EXIT:
            printf("User '%s' exited.\n", msg->sender_name);
            user_list_remove(comm->user_list, source_ip, source_port);
            break;

        // 其他命令...
        default:
            break;
    }
}


/*
 * 函数定义：communicator_send_ans_entry
 * 功能：向指定用户单播发送一条“响应上线”消息。
 */
void communicator_send_ans_entry(communicator_t* comm, const char* dest_ip, uint16_t dest_port)
{
    char buffer[MAX_PACKET_LEN] = {0};
    // 此处需要一个新的 build_ans_entry_packet 函数，我们暂时用 build_entry_packet 代替，
    // 但你应该在 protocol_parser.c 中创建它。为简化，我们先跳过。
    int packet_len = build_entry_packet(buffer, sizeof(buffer), comm->packet_counter++, comm->user_name, comm->host_name);
    
    if(packet_len < 0) return;

    struct sockaddr_in dest_addr;
    memset(&dest_addr, 0, sizeof(dest_addr));
    dest_addr.sin_family = AF_INET;
    dest_addr.sin_port = htons(dest_port);
    inet_pton(AF_INET, dest_ip, &dest_addr.sin_addr); // 将字符串IP转为网络格式

    sendto(comm->sock, buffer, packet_len, 0, (struct sockaddr*)&dest_addr, sizeof(dest_addr));
    printf("Sent answer to %s:%u\n", dest_ip, dest_port);
}
```

**操作指南：** 打开 `src/model/user.c` 并用以下代码替换其内容。

C

```
#include "model/user.h"
#include <stdlib.h>
#include <string.h>

int user_list_init(user_list_t* list) {
    if (!list) return -1;
    list->head = NULL;
    list->count = 0;
    InitializeCriticalSection(&list->lock);
    return 0;
}

void user_list_destroy(user_list_t* list) {
    if (!list) return;
    user_list_clear(list); // 清理所有节点
    DeleteCriticalSection(&list->lock);
}

// 线程安全地添加或更新用户
void user_list_add(user_list_t* list, const user_node_t* user_data) {
    if (!list || !user_data) return;

    EnterCriticalSection(&list->lock);

    // 检查用户是否已存在
    user_node_t* current = list->head;
    while (current != NULL) {
        if (strcmp(current->ipAddress, user_data->ipAddress) == 0 && current->port == user_data->port) {
            // 用户已存在，更新信息即可
            strcpy_s(current->userName, sizeof(current->userName), user_data->userName);
            strcpy_s(current->hostName, sizeof(current->hostName), user_data->hostName);
            LeaveCriticalSection(&list->lock);
            return; // 完成更新，直接返回
        }
        current = current->next;
    }

    // 用户不存在，创建新节点并插入到链表头部
    user_node_t* new_node = (user_node_t*)malloc(sizeof(user_node_t));
    if (new_node) {
        memcpy(new_node, user_data, sizeof(user_node_t));
        new_node->next = list->head;
        list->head = new_node;
        list->count++;
    }

    LeaveCriticalSection(&list->lock);
}

// 线程安全地移除用户
void user_list_remove(user_list_t* list, const char* ip_address, uint16_t port) {
    if (!list || !ip_address) return;

    EnterCriticalSection(&list->lock);

    user_node_t* current = list->head;
    user_node_t* prev = NULL;

    while (current != NULL) {
        if (strcmp(current->ipAddress, ip_address) == 0 && current->port == port) {
            // 找到了要删除的节点
            if (prev == NULL) { // 如果是头节点
                list->head = current->next;
            } else {
                prev->next = current->next;
            }
            free(current); // 释放内存
            list->count--;
            break; // 完成删除
        }
        prev = current;
        current = current->next;
    }

    LeaveCriticalSection(&list->lock);
}

void user_list_clear(user_list_t* list) {
    if(!list) return;
    EnterCriticalSection(&list->lock);
    user_node_t* current = list->head;
    while(current != NULL) {
        user_node_t* next = current->next;
        free(current);
        current = next;
    }
    list->head = NULL;
    list->count = 0;
    LeaveCriticalSection(&list->lock);
}
```

### **如何调试与预期结果**

现在你的程序已经具备了收发和处理上线消息的能力。

#### **调试步骤：“自言自语”测试**

最简单的测试方法就是让程序接收自己发送的广播包。

1. **设置断点**：
    - 在 `communicator.c` 的 `process_packet` 函数的第一行（`printf("Received command..."`）设置一个断点。
2. **启动调试**：
    - 为调试会话**设置好程序参数**（例如 `Alice`）。
    - 点击**调试(🐞)按钮**启动程序。

#### **预期结果**

1. **程序启动后，几乎立刻就会在 `process_packet` 的断点处暂停！**
2. **为什么？**
    - `main` 函数调用 `communicator_broadcast_entry`，向局域网发送了上线广播。
    - `communicator_listen_loop` 线程几乎同时收到了这个由自己发出的广播包。
    - `recvfrom` 接收到数据，调用 `parse_message`，然后调用 `process_packet`，最终命中断点。
3. **在断点处观察**：
    - 查看 `msg` 结构体，它的 `command` 应该是 `IPMSG_BR_ENTRY`，`sender_name` 应该是 `"Alice"`。
    - 查看 `source_ip`，它可能是你的本机IP地址（如 `192.168.1.x`）或者是环回地址 `127.0.0.1`。
    - 按 `F8` (Step Over) 单步执行，你会看到程序打印出 `"User 'Alice' entered."` 等日志。

**最终测试：双实例互动** 如果你在**两台电脑**上（或在一台电脑上打开两个独立的命令行窗口）分别运行 `ipmsg_client.exe Alice` 和 `ipmsg_client.exe Bob`，你将会看到：

- Bob 的窗口会打印出 "User 'Alice' entered."
- Alice 的窗口会打印出 "User 'Bob' entered."
- 紧接着，Bob 的窗口会打印出 "Received answer from 'Alice'."，因为 Alice 对 Bob 的上线包做出了响应。

如果能实现双实例的互动，那么恭喜你，你已经完成了整个项目的核心通信回路！
