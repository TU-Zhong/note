这个结构将源代码按功能（核心、模型、UI）分入不同目录，这是 C 项目中管理复杂度的良好实践。

### **1. 项目结构**

```
IPMsg-Lite-Client-C/
├── CMakeLists.txt              # 项目编译配置文件
│
└── src/                        # 所有源代码
    ├── main.c                  # 程序主入口
    │
    ├── core/                   # 核心功能模块
    │   ├── protocol.h          # 协议相关的常量和枚举
    │   ├── protocol_parser.h   # 协议封包和解包工具
    │   ├── communicator.h      # UDP 通信管理器
    │   └── file_transfer.h     # TCP 文件传输管理器
    │
    ├── model/                  # 数据模型模块
    │   └── user.h              # 用户及用户列表的数据结构和管理函数
    │
    └── ui/                     # 用户界面模块
        └── console.h           # 命令行界面的定义和函数
```

_(注意: `.c` 文件将与对应的 `.h` 文件位于同一目录，此处为简洁省略)_

### **2. 头文件实现**

以下是与上述结构对应的所有头文件的具体实现。

#### **`src/model/user.h`**

此文件定义了用户和用户列表的数据结构及相关操作。

C

```
#ifndef USER_H
#define USER_H

#include <stdint.h>
#include <windows.h> // For CRITICAL_SECTION

#define MAX_NAME_LEN 64
#define MAX_HOSTNAME_LEN 64
#define MAX_IP_LEN 16

// 描述单个用户的链表节点
typedef struct UserNode {
    char userName[MAX_NAME_LEN];
    char groupName[MAX_HOSTNAME_LEN];
    char hostName[MAX_HOSTNAME_LEN];
    char ipAddress[MAX_IP_LEN];
    uint16_t port;
    struct UserNode* next;
} user_node_t;

// 线程安全的用户列表管理器
typedef struct {
    user_node_t* head;
    int count;
    CRITICAL_SECTION lock; // 用于保证链表操作的线程安全
} user_list_t;

// --- 函数定义 ---

// 初始化和销毁用户列表
int user_list_init(user_list_t* list);
void user_list_destroy(user_list_t* list);

// 添加和移除用户
void user_list_add(user_list_t* list, const user_node_t* user_data);
void user_list_remove(user_list_t* list, const char* ip_address, uint16_t port);

// 查询用户
user_node_t* user_list_find_by_index(user_list_t* list, int index);

// 复制所有用户信息到一个数组中（用于显示）
int user_list_get_all(user_list_t* list, user_node_t dest_array[], int max_count);

#endif //USER_H
```

#### **`src/core/protocol.h`**

定义 IP Messenger 协议的常量。

C

```
#ifndef PROTOCOL_H
#define PROTOCOL_H

#include <stdint.h>

// 协议版本和默认端口
#define IPMSG_PROTOCOL_VERSION 1
#define IPMSG_DEFAULT_PORT 2425

// 协议核心命令
typedef enum {
    IPMSG_NOOP           = 0,
    IPMSG_BR_ENTRY       = 1,  // 用户上线
    IPMSG_BR_EXIT        = 2,  // 用户下线
    IPMSG_ANSENTRY       = 3,  // 响应上线
    IPMSG_SENDMSG        = 32, // 发送消息
} ipmsg_command_t;

// 命令选项
#define IPMSG_FILEATTACHOPT 0x00200000 // 文件附件选项

#endif //PROTOCOL_H
```

#### **`src/core/protocol_parser.h`**

定义协议解包和封包的工具函数。

C

```
#ifndef PROTOCOL_PARSER_H
#define PROTOCOL_PARSER_H

#include <stdint.h>
#include "../model/user.h" // 需要用户定义中的宏

#define MAX_PACKET_LEN 4096
#define MAX_ADDITIONAL_LEN 2048

// 解析后的消息结构
typedef struct {
    int version;
    uint64_t packet_no;
    char sender_name[MAX_NAME_LEN];
    char sender_host[MAX_HOSTNAME_LEN];
    uint32_t command;
    uint32_t option;
    char additional_section[MAX_ADDITIONAL_LEN];
} parsed_message_t;

// --- 函数定义 ---

// 构造不同类型的IPMsg数据包，返回包的实际长度
int build_entry_packet(char* buffer, size_t buffer_size, uint64_t packet_no, const char* user_name, const char* host_name);
int build_exit_packet(char* buffer, size_t buffer_size, uint64_t packet_no, const char* user_name, const char* host_name);

// 解析收到的数据包，成功返回0，失败返回-1
int parse_message(const char* packet, size_t packet_len, parsed_message_t* out_msg);

#endif //PROTOCOL_PARSER_H
```

#### **`src/core/communicator.h`**

负责 UDP 通信的核心模块。

C

```
#ifndef COMMUNICATOR_H
#define COMMUNICATOR_H

#include <winsock2.h>
#include <windows.h>
#include "../model/user.h"

// 前向声明，避免头文件循环依赖
struct Console;

// UDP通信器结构
typedef struct {
    SOCKET sock;
    HANDLE listen_thread_handle;
    volatile int is_running;
    volatile uint64_t packet_counter;

    char user_name[MAX_NAME_LEN];
    char host_name[MAX_HOSTNAME_LEN];

    // 依赖项，通过指针持有
    user_list_t* user_list;
    struct Console* console;
} communicator_t;

// --- 函数定义 ---

// 初始化和销毁通信器
int communicator_init(communicator_t* comm, user_list_t* user_list, struct Console* console, const char* user_name);
void communicator_destroy(communicator_t* comm);

// 发送广播和单播消息
void communicator_broadcast_entry(communicator_t* comm);
void communicator_broadcast_exit(communicator_t* comm);
void communicator_send_ans_entry(communicator_t* comm, const char* dest_ip, uint16_t dest_port);

#endif //COMMUNICATOR_H
```

#### **`src/ui/console.h`**

负责命令行交互的模块。

C

```
#ifndef CONSOLE_H
#define CONSOLE_H

#include <windows.h>
#include "../model/user.h"

// 命令回调函数指针
// context是传递给回调的上下文指针，通常是主程序结构体
typedef void (*command_callback_t)(void* context, int argc, char* argv[]);

// 命令行UI结构
typedef struct Console {
    CRITICAL_SECTION print_lock; // 保证多线程打印到控制台时内容不交错
    volatile int should_exit;

    // 命令处理回调
    command_callback_t list_users_cb;
    command_callback_t send_file_cb;
    command_callback_t exit_app_cb;
    void* callback_context;
} console_t;

// --- 函数定义 ---

// 初始化和销毁
int console_init(console_t* console);
void console_destroy(console_t* console);

// 注册命令回调函数
void console_register_callbacks(console_t* console, void* context, command_callback_t list_cb, command_callback_t send_cb, command_callback_t exit_cb);

// 启动UI的阻塞式输入循环
void console_run(console_t* console);

// 供其他线程调用的线程安全打印函数
void console_print_message(console_t* console, const char* message);
void console_display_users(console_t* console, user_list_t* user_list);

#endif //CONSOLE_H
```

### **3. `CMakeLists.txt` 和编译**

#### **`CMakeLists.txt`**

将此文件放在项目根目录。

CMake

```
cmake_minimum_required(VERSION 3.15)
project(IPMsg-Lite-Client-C C)

set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)

# 添加头文件根目录
include_directories(src)

# 添加所有源文件（.c）来生成可执行文件
# 注意：你需要在这些 .c 文件中提供函数的具体实现
add_executable(ipmsg_client
    src/main.c
    src/core/communicator.c
    src/core/protocol_parser.c
    src/core/file_transfer.c
    src/model/user.c
    src/ui/console.c
)

# 仅在Windows平台上链接Winsock库
if(WIN32)
    target_link_libraries(ipmsg_client ws2_32)
endif()
```

#### **编译命令**

编译步骤与之前提供的一致，CLion 可以无缝处理。

1. **CLion 内:** 打开项目文件夹，点击 `Build` (锤子图标)。
2. **手动命令行:**
    
    DOS
    
    ```
    cd path\to\IPMsg-Lite-Client-C
    mkdir build
    cd build
    cmake ..
    cmake --build .
    ```
    
    编译后的 `ipmsg_client.exe` 会在 `build\Debug` 目录中。

### **4. 测试步骤**

测试流程与之前相同，因为这只涉及程序的外部行为。

1. **启动实例 A:** 在命令行中运行 `ipmsg_client.exe Alice`。
2. **启动实例 B:** 在另一命令行中运行 `ipmsg_client.exe Bob`。
3. **验证上线:** 检查双方窗口是否都收到了对方的上线通知。
4. **测试列表:** 在任一窗口输入 `list` 并回车，验证是否能看到对方信息。
5. **测试下线:** 在任一窗口输入 `exit`，验证该程序关闭，且另一方收到下线通知。
