import java.time.YearMonth;
import java.util.*;
import java.util.regex.Pattern;

// 支付处理器接口
interface PaymentProcessor {
    boolean validate() throws PaymentException;
    String processPayment(double amount);
    String getPaymentType();
}

// 自定义异常
class PaymentException extends Exception {
    private final String errorCode;

    public PaymentException(String errorCode, String message) {
        super(message);
        this.errorCode = errorCode;
    }

    public String getErrorCode() {
        return errorCode;
    }
}

// 工厂类
class PaymentProcessorFactory {
    public static PaymentProcessor createProcessor(String type, Map<String, String> params) {
        return switch (type.toUpperCase()) {
            case "CREDIT_CARD" -> new CreditCardProcessor(
                    params.get("cardNumber"),
                    params.get("cardHolder"),
                    params.get("expiryDate"),
                    params.get("cvv"));
            case "PAYPAL" -> new PayPalProcessor(
                    params.get("email"),
                    params.get("password"));
            case "CRYPTO" -> new CryptoProcessor(
                    params.get("walletAddress"),
                    params.get("privateKey"));
            default -> throw new IllegalArgumentException("不支持的支付类型");
        };
    }
}

// 信用卡支付实现
class CreditCardProcessor implements PaymentProcessor {
    private final String cardNumber;
    private final String cardHolder;
    private final String expiryDate;
    private final String cvv;

    public CreditCardProcessor(String cardNumber, String cardHolder, 
                              String expiryDate, String cvv) {
        this.cardNumber = cardNumber;
        this.cardHolder = cardHolder;
        this.expiryDate = expiryDate;
        this.cvv = cvv;
    }

    @Override
    public boolean validate() throws PaymentException {
        // 验证有效期格式 MM/yy
        if (!expiryDate.matches("(0[1-9]|1[0-2])/[0-9]{2}")) {
            throw new PaymentException("CARD_001", "无效的有效期格式");
        }

        // 检查是否过期
        YearMonth expiry = YearMonth.parse("20" + expiryDate.substring(3) + "-" + expiryDate.substring(0,2));
        if (expiry.isBefore(YearMonth.now())) {
            throw new PaymentException("CARD_002", "信用卡已过期");
        }

        // 验证CVV
        if (cvv == null || !cvv.matches("\\d{3,4}")) {
            throw new PaymentException("CARD_003", "无效的CVV码");
        }

        return true;
    }

    @Override
    public String processPayment(double amount) {
        return "CARD-" + UUID.randomUUID().toString().substring(0, 12).toUpperCase();
    }

    @Override
    public String getPaymentType() {
        return "信用卡支付";
    }
}

// PayPal支付实现
class PayPalProcessor implements PaymentProcessor {
    private static final Pattern EMAIL_REGEX = 
        Pattern.compile("^[A-Za-z0-9+_.-]+@(.+)$");
    
    private final String email;
    private final String password;

    public PayPalProcessor(String email, String password) {
        this.email = email;
        this.password = password;
    }

    @Override
    public boolean validate() throws PaymentException {
        if (!EMAIL_REGEX.matcher(email).matches()) {
            throw new PaymentException("PAYPAL_001", "邮箱格式不正确");
        }
        
        if (password == null || password.length() < 6) {
            throw new PaymentException("PAYPAL_002", "密码至少需要6位");
        }
        
        return true;
    }

    @Override
    public String processPayment(double amount) {
        return "PAYPAL-" + UUID.randomUUID().toString().substring(0, 8).toUpperCase();
    }

    @Override
    public String getPaymentType() {
        return "PayPal支付";
    }
}

// 加密货币支付实现
class CryptoProcessor implements PaymentProcessor {
    private final String walletAddress;
    private final String privateKey;

    public CryptoProcessor(String walletAddress, String privateKey) {
        this.walletAddress = walletAddress;
        this.privateKey = privateKey;
    }

    @Override
    public boolean validate() throws PaymentException {
        if (walletAddress == null || !walletAddress.startsWith("0x") || walletAddress.length() != 42) {
            throw new PaymentException("CRYPTO_001", "无效的钱包地址");
        }
        
        if (privateKey == null || privateKey.length() < 64) {
            throw new PaymentException("CRYPTO_002", "私钥格式错误");
        }
        
        return true;
    }

    @Override
    public String processPayment(double amount) {
        return "CRYPTO-" + UUID.randomUUID().toString().substring(0, 16).toUpperCase();
    }

    @Override
    public String getPaymentType() {
        return "加密货币支付";
    }
}

// 演示程序
public class PaymentDemo {
    private static final Scanner scanner = new Scanner(System.in);

    public static void main(String[] args) {
        while (true) {
            try {
                System.out.println("\n=== 支付系统演示 ===");
                System.out.println("1. 信用卡支付");
                System.out.println("2. PayPal支付");
                System.out.println("3. 加密货币支付");
                System.out.println("4. 退出");
                System.out.print("请选择支付方式：");
                
                int choice = Integer.parseInt(scanner.nextLine());
                if (choice == 4) break;

                Map<String, String> params = new HashMap<>();
                String type = "";
                
                switch (choice) {
                    case 1:
                        type = "CREDIT_CARD";
                        System.out.print("请输入卡号：");
                        params.put("cardNumber", scanner.nextLine());
                        System.out.print("持卡人姓名：");
                        params.put("cardHolder", scanner.nextLine());
                        System.out.print("有效期(MM/yy)：");
                        params.put("expiryDate", scanner.nextLine());
                        System.out.print("CVV：");
                        params.put("cvv", scanner.nextLine());
                        break;
                        
                    case 2:
                        type = "PAYPAL";
                        System.out.print("PayPal邮箱：");
                        params.put("email", scanner.nextLine());
                        System.out.print("密码：");
                        params.put("password", scanner.nextLine());
                        break;
                        
                    case 3:
                        type = "CRYPTO";
                        System.out.print("钱包地址：");
                        params.put("walletAddress", scanner.nextLine());
                        System.out.print("私钥：");
                        params.put("privateKey", scanner.nextLine());
                        break;
                        
                    default:
                        System.out.println("无效的选择");
                        continue;
                }

                PaymentProcessor processor = PaymentProcessorFactory.createProcessor(type, params);
                if (processor.validate()) {
                    System.out.print("请输入支付金额：");
                    double amount = Double.parseDouble(scanner.nextLine());
                    String transactionId = processor.processPayment(amount);
                    System.out.printf("支付成功！[%s] 交易ID：%s%n", 
                            processor.getPaymentType(), transactionId);
                }
            } catch (PaymentException e) {
                System.out.printf("验证失败：[%s] %s%n", e.getErrorCode(), e.getMessage());
            } catch (Exception e) {
                System.out.println("发生错误：" + e.getMessage());
            }
        }
        System.out.println("感谢使用支付系统！");
    }
}

2. **验证逻辑亮点**：
   - 信用卡有效期使用`YearMonth`处理日期比较
   - 正则表达式验证邮箱和地址格式
   - 详细的错误代码定义

3. **交互设计**：
   - 循环操作界面
   - 统一的错误处理机制
   - 类型安全的参数输入

4. **扩展性设计**：
   - 每个处理器独立实现验证逻辑
   - 统一接口保证系统可扩展性
