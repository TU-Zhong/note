```java
import java.util.Stack;
import java.util.regex.Pattern;
import java.util.regex.Matcher;

public class ExpressionValidator {
    
    public static boolean isValidExpression(String expr) {
        // 预处理
        String expression = expr.replaceAll("\\s+", "").toLowerCase();
        if (expression.isEmpty()) return false;

        // 1. 基础字符检查
        Pattern allowedChars = Pattern.compile("^[\\d+\\-*/^!().]+$");
        boolean hasValidChars = allowedChars.matcher(expression).matches();
        boolean hasFunctions = Pattern.compile("sin|cos|ln").matcher(expression).find();
        if (!hasValidChars && !hasFunctions) {
            return false;
        }

        // 2. 括号匹配检查
        Stack<Character> stack = new Stack<>();
        for (char c : expression.toCharArray()) {
            if (c == '(') {
                stack.push(c);
            } else if (c == ')') {
                if (stack.isEmpty()) return false;
                stack.pop();
            }
        }
        if (!stack.isEmpty()) return false;

        // 3. 函数调用验证
        Pattern funcPattern = Pattern.compile("(sin|cos|ln)(?![(])");
        if (funcPattern.matcher(expression).find()) {
            return false;
        }

        // 4. 运算符连续检查
        Pattern operatorPattern = Pattern.compile("(?<![\\d)])[+*/^]|[-+*/^]{2,}");
        if (operatorPattern.matcher(expression).find()) {
            return false;
        }

        // 5. 小数点验证
        Pattern decimalPattern = Pattern.compile("\\.\\d*\\.|^\\.|\\d+\\.\\d+\\.");
        if (decimalPattern.matcher(expression).find()) {
            return false;
        }

        // 6. 阶乘符号验证
        Pattern factorialPattern = Pattern.compile("(?<![!\\d)])!");
        if (factorialPattern.matcher(expression).find()) {
            return false;
        }

        // 7. 首尾字符验证
        Pattern startsWith = Pattern.compile("^[\\d(sincosln-]");
        Pattern endsWith = Pattern.compile("[\\d)!]$");
        if (!startsWith.matcher(expression).find() || 
            !endsWith.matcher(expression).find()) {
            return false;
        }

        return true;
    }

    public static void main(String[] args) {
        // 测试用例
        System.out.println(isValidExpression("sin(3)+cos(5!)"));   // true
        System.out.println(isValidExpression("2.5*(ln(-4)+3)"));  // true
        System.out.println(isValidExpression("5!+*6"));           // false
        System.out.println(isValidExpression("sinx+5"));          // false
        System.out.println(isValidExpression("3.14.15"));         // false
        System.out.println(isValidExpression("(2+3))*4"));        // false
    }
}
```

### 关键实现说明：

1. **正则表达式调整**：
   - Java中的正则表达式需要用双反斜杠转义（`\\`代替`\`）
   - 用`Pattern`和`Matcher`类替代Python的`re`模块

2. **字符串预处理**：
   ```java
   String expression = expr.replaceAll("\\s+", "").toLowerCase();
   ```

3. **函数验证逻辑**：
   ```java
   Pattern funcPattern = Pattern.compile("(sin|cos|ln)(?![(])");
   if (funcPattern.matcher(expression).find()) {
       return false;
   }
   ```

4. **负号特殊处理**：
   - 允许类似`-5`和`3*(-4)`的情况
   - 通过正则`(?<![\\d)])[+*/^]`确保运算符合法

5. **阶乘验证**：
   ```java
   Pattern factorialPattern = Pattern.compile("(?<![!\\d)])!");
   ```

6. **首尾字符验证**：
   ```java
   Pattern startsWith = Pattern.compile("^[\\d(sincosln-]");
   Pattern endsWith = Pattern.compile("[\\d)!]$");
   ```

1. 如果需要支持更多函数（如`tan`、`sqrt`等），修改`funcPattern`的正则表达式
2. 要处理科学计数法（如`1.2e3`），添加对应的正则验证
3. 对于更复杂的表达式解析，建议使用递归下降解析器或ANTLR等工具
