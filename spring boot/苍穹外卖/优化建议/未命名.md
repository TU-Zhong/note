### 1. 修改DishServiceImpl实现缓存
```java
package com.sky.service.impl;

import com.sky.constant.StatusConstant;
import com.sky.context.BaseContext;
import com.sky.entity.Dish;
import com.sky.entity.DishFlavor;
import com.sky.exception.BusinessException;
import com.sky.mapper.DishMapper;
import com.sky.mapper.DishFlavorMapper;
import com.sky.service.DishService;
import com.sky.vo.DishVO;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.BeanUtils;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.time.LocalDateTime;
import java.util.List;
import java.util.concurrent.TimeUnit;

@Service
@Slf4j
public class DishServiceImpl implements DishService {
    @Autowired
    DishMapper dishMapper;
    @Autowired
    DishFlavorMapper dishFlavorMapper;
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    // Redis缓存key前缀
    private static final String DISH_KEY = "dish:";
    private static final String DISH_LIST_KEY = "dish:list:category:";
    private static final String DISH_PAGE_KEY = "dish:page:";
    // 缓存过期时间（24小时）
    private static final long CACHE_EXPIRE_TIME = 24;
    
    @Override
    public DishVO findDishWithFlavor(Integer id) {
        // 构建缓存key
        String key = DISH_KEY + id;
        
        // 查询Redis缓存
        DishVO dishVO = (DishVO) redisTemplate.opsForValue().get(key);
        
        // 如果缓存中存在，直接返回
        if (dishVO != null) {
            log.info("从缓存中获取菜品详情, id: {}", id);
            return dishVO;
        }
        
        // 缓存中不存在，查询数据库
        log.info("从数据库中获取菜品详情, id: {}", id);
        dishVO = dishMapper.findDishByIdWithFlavor(id);
        
        // 将查询结果存入缓存
        if (dishVO != null) {
            redisTemplate.opsForValue().set(key, dishVO, CACHE_EXPIRE_TIME, TimeUnit.HOURS);
        }
        
        return dishVO;
    }

    @Override
    public List<Dish> findDishesByCategoryId(Integer categoryId) {
        // 构建缓存key
        String key = DISH_LIST_KEY + categoryId;
        
        // 查询Redis缓存
        List<Dish> dishes = (List<Dish>) redisTemplate.opsForValue().get(key);
        
        // 如果缓存中存在，直接返回
        if (dishes != null && !dishes.isEmpty()) {
            log.info("从缓存中获取分类菜品列表, categoryId: {}", categoryId);
            return dishes;
        }
        
        // 缓存中不存在，查询数据库
        log.info("从数据库中获取分类菜品列表, categoryId: {}", categoryId);
        dishes = dishMapper.findDishesByCategoryId(categoryId);
        
        // 将查询结果存入缓存
        if (dishes != null && !dishes.isEmpty()) {
            redisTemplate.opsForValue().set(key, dishes, CACHE_EXPIRE_TIME, TimeUnit.HOURS);
        }
        
        return dishes;
    }

    @Override
    public List<Dish> page(String categoryId, String name, String status, Integer offset, Integer limit) {
        // 构建缓存key - 将查询条件组合成key
        String key = DISH_PAGE_KEY + "c:" + (categoryId == null ? "" : categoryId) + 
                     ":n:" + (name == null ? "" : name) + 
                     ":s:" + (status == null ? "" : status) + 
                     ":o:" + offset + ":l:" + limit;
        
        // 查询Redis缓存
        List<Dish> dishes = (List<Dish>) redisTemplate.opsForValue().get(key);
        
        // 如果缓存中存在，直接返回
        if (dishes != null && !dishes.isEmpty()) {
            log.info("从缓存中获取菜品分页数据");
            return dishes;
        }
        
        // 缓存中不存在，查询数据库
        log.info("从数据库中获取菜品分页数据");
        dishes = dishMapper.page(categoryId, name, status, offset, limit);
        
        // 将查询结果存入缓存
        if (dishes != null && !dishes.isEmpty()) {
            redisTemplate.opsForValue().set(key, dishes, CACHE_EXPIRE_TIME, TimeUnit.HOURS);
        }
        
        return dishes;
    }

    @Override
    public Dish findById(int id) {
        // 构建缓存key
        String key = DISH_KEY + "simple:" + id;
        
        // 查询Redis缓存
        Dish dish = (Dish) redisTemplate.opsForValue().get(key);
        
        // 如果缓存中存在，直接返回
        if (dish != null) {
            log.info("从缓存中获取菜品基本信息, id: {}", id);
            return dish;
        }
        
        // 缓存中不存在，查询数据库
        log.info("从数据库中获取菜品基本信息, id: {}", id);
        dish = dishMapper.findById(id);
        
        // 将查询结果存入缓存
        if (dish != null) {
            redisTemplate.opsForValue().set(key, dish, CACHE_EXPIRE_TIME, TimeUnit.HOURS);
        }
        
        return dish;
    }

    @Override
    @Transactional
    public void deleteById(int id) {
        dishMapper.deleteById(id);
        
        // 清除相关缓存
        clearDishCache(id);
    }

    @Override
    @Transactional
    public void save(DishVO dishVO) {
        //1.核心字段验证
        if(dishVO.getName()==null || dishVO.getName().trim().isEmpty()){
            throw new BusinessException("菜品名不能为空");
        }
        if(dishVO.getImage()==null || dishVO.getImage().trim().isEmpty()){
            throw new BusinessException("图片不能为空");
        }
        if(dishVO.getCategoryId()==null){
            throw new BusinessException("菜品分类不能为空");
        }
        if(dishVO.getPrice()==null){
            throw new BusinessException("价格不能为空");
        }
        //2. 设置默认值
        Dish dish=new Dish();
        BeanUtils.copyProperties(dishVO, dish);
        if(dish.getStatus()==null){
            dish.setStatus(StatusConstant.DISABLE);
        }
        if(dish.getDescription()==null){
            dish.setDescription("");
        }
        //3. 设置审计字段
        LocalDateTime now=LocalDateTime.now();
        dish.setCreateTime(now);
        dish.setUpdateTime(now);
        Long currentUserId= BaseContext.getCurrentId();
        dish.setCreateUser(currentUserId);
        dish.setUpdateUser(currentUserId);

        //执行插入
        dishMapper.insert(dish);
        Long id=dish.getId();
        List<DishFlavor> flavors=dishVO.getFlavors();
        if(flavors!=null && !flavors.isEmpty()){
            flavors.forEach(flavor->{flavor.setDishId(id);});
            dishFlavorMapper.insertBatch(flavors);
        }
        
        // 清除相关分类的菜品列表缓存
        clearCategoryDishListCache(dish.getCategoryId());
    }
    
    /**
     * 清除菜品相关的所有缓存
     * @param id 菜品ID
     */
    private void clearDishCache(long id) {
        // 获取菜品信息，以便清除分类相关缓存
        Dish dish = dishMapper.findById((int)id);
        if (dish != null) {
            // 清除分类相关缓存
            clearCategoryDishListCache(dish.getCategoryId());
        }
        
        // 清除菜品详情缓存
        String dishKey = DISH_KEY + id;
        redisTemplate.delete(dishKey);
        
        // 清除菜品基本信息缓存
        String simpleKey = DISH_KEY + "simple:" + id;
        redisTemplate.delete(simpleKey);
        
        // 清除所有分页查询缓存
        clearAllPageCache();
        
        log.info("清除菜品ID {} 相关的缓存", id);
    }
    
    /**
     * 清除分类下的菜品列表缓存
     * @param categoryId 分类ID
     */
    private void clearCategoryDishListCache(Long categoryId) {
        if (categoryId != null) {
            String key = DISH_LIST_KEY + categoryId;
            redisTemplate.delete(key);
            log.info("清除分类 {} 的菜品列表缓存", categoryId);
        }
    }
    
    /**
     * 清除所有分页查询缓存
     */
    private void clearAllPageCache() {
        Set<String> keys = redisTemplate.keys(DISH_PAGE_KEY + "*");
        if (keys != null && !keys.isEmpty()) {
            redisTemplate.delete(keys);
            log.info("清除所有菜品分页查询缓存");
        }
    }
}
```

### 4. 在DishServiceImpl中实现新方法
```java
@Override
@Transactional
public void update(DishVO dishVO) {
    // 1. 更新菜品基本信息
    Dish dish = new Dish();
    BeanUtils.copyProperties(dishVO, dish);
    
    // 设置更新时间和更新人
    dish.setUpdateTime(LocalDateTime.now());
    dish.setUpdateUser(BaseContext.getCurrentId());
    
    // 更新菜品信息
    dishMapper.update(dish);
    
    // 2. 更新口味信息 - 先删除原有口味，再添加新口味
    Long dishId = dish.getId();
    
    // 删除原有口味
    dishFlavorMapper.deleteByDishId(dishId);
    
    // 添加新口味
    List<DishFlavor> flavors = dishVO.getFlavors();
    if (flavors != null && !flavors.isEmpty()) {
        flavors.forEach(flavor -> flavor.setDishId(dishId));
        dishFlavorMapper.insertBatch(flavors);
    }
    
    // 3. 清除相关缓存
    clearDishCache(dishId);
}

@Override
public void startOrStop(Integer status, Long id) {
    Dish dish = Dish.builder()
            .id(id)
            .status(status)
            .updateTime(LocalDateTime.now())
          
			.updateUser(BaseContext.getCurrentId())
            .build();
    
    dishMapper.update(dish);
    
    // 清除相关缓存
    clearDishCache(id);
}
```

### 缓存策略说明

1. **缓存内容**：
    - 菜品详情（带口味）：`dish:{id}`
    - 菜品基本信息：`dish:simple:{id}`
    - 分类下的菜品列表：`dish:list:category:{categoryId}`
    - 分页查询结果：`dish:page:c:{categoryId}:n:{name}:s:{status}:o:{offset}:l:{limit}`
2. **缓存时间**：
    - 设置为24小时，可以根据实际需求调整
3. **缓存更新策略**：
    - 当新增、修改、删除、上下架菜品时，清除相关缓存
    - 使用专门的方法清除不同类型的缓存：
        - `clearDishCache(id)` - 清除单个菜品的所有相关缓存
        - `clearCategoryDishListCache(categoryId)` - 清除分类下的菜品列表缓存
        - `clearAllPageCache()` - 清除所有分页查询缓存
4. **缓存查询逻辑**：
    - 先查缓存，缓存命中则直接返回
    - 缓存未命中则查询数据库，并将结果存入缓存

### 优化建议

1. **批量操作优化**：
    - 对于批量删除或状态变更操作，可以一次性清除多个缓存
2. **缓存预热**：
    - 可以在应用启动时预先加载热门分类的菜品数据到缓存中
3. **缓存穿透防护**：
    - 对于不存在的数据，可以缓存空值或使用布隆过滤器
4. **分布式锁**：
    - 在高并发环境下，可以使用分布式锁防止缓存击穿
5. **监控缓存**：
    - 添加缓存命中率监控，以便调整缓存策略

这样实现后，菜品列表和详情信息将会被缓存到Redis中，大大减少数据库访问，提高系统性能。特别是对于按分类查询菜品列表这种高频操作，缓存优化效果会非常明显。
