#include <cstdint>
#include <vector>

class HyperLogLog {
private:
    uint8_t p; 
    uint32_t m; 
    std::vector<uint8_t> buckets; 

public:
    HyperLogLog(uint8_t precision = 14) : p(precision), m(1 << precision) {
        buckets.assign(m, 0); 
    }

    // 核心的 Add 方法（你刚刚学会的位运算魔法！）
    void add(uint64_t hash) {
        uint32_t index = hash >> (64 - p);
        uint64_t w = hash << p;

        uint8_t rank = 0;
        if (w == 0) {
            rank = (64 - p) + 1; 
        } else {
            rank = __builtin_clzll(w) + 1;
        }

        if (rank > buckets[index]) {
            buckets[index] = rank;
        }
        
        // 打印一下日志，看看底层发生了什么
        std::cout << "[底层日志] 收到哈希: " << hash 
                  << " -> 算得桶索引(Index): " << index 
                  << " -> 算得连续零(Rank): " << (int)rank << std::endl;
