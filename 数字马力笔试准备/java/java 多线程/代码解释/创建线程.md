package ManyThreads;  
  
import java.util.concurrent.*;  
  
public class Create {  
    public static void main(String[] args) throws InterruptedException {  
        //1.继承Thread类  
        ThreadDemo1 thread1=new ThreadDemo1();  
        thread1.start();  
        //2.实现Runnable接口  
        ThreadDemo2 runnable=new ThreadDemo2();  
        Thread thread2=new Thread(runnable);  
        thread2.start();  
        //3.匿名内部类  
        Thread thread3=new Thread(new Runnable() {  
            @Override  
            public void run() {  
                System.out.println("3.匿名内部类创建线程："+Thread.currentThread().getName());  
            }  
        });  
        thread3.start();  
        //4.Lambda表达式  
        Thread thread4=new Thread(()->{  
            System.out.println("通过Lambda表达式创建线程："+Thread.currentThread().getName());  
        });  
        thread4.start();  
        //5.实现Callable接口  
        FutureTask<String> futureTask=new FutureTask<>(new ThreadDemo5());  
        Thread thread5=new Thread(futureTask);  
        thread5.start();  
        try{  
            String result=futureTask.get();  
            System.out.println("5.Callable返回值："+result);  
        }catch (InterruptedException | ExecutionException e){  
            e.printStackTrace();  
        }  
        //6.使用线程池  
        ExecutorService executorService= Executors.newFixedThreadPool(2);  
        try{  
            executorService.execute(()-> System.out.println("6. 线程池创建线程："+Thread.currentThread().getName()));  
            Future<String> future=executorService.submit(()->"任务执行结果");  
        }finally {  
            //关闭线程池  
            executorService.shutdown();  
            executorService.awaitTermination(5,TimeUnit.SECONDS);  
        }  
        //7.使用CompletableFuture  
        CompletableFuture<Void> future = CompletableFuture.runAsync(() -> System.out.println("7. CompletableFuture创建线程：" + Thread.currentThread().getName()));  
        future.join();  
    }  
  
}  
class ThreadDemo1 extends Thread{  
    @Override  
    public void run(){  
        System.out.println("1.继承Thread类创建线程："+Thread.currentThread().getName());  
    }  
}  
class ThreadDemo2 implements Runnable {  
    @Override  
    public void run() {  
        System.out.println("2.实现Runnable接口创建线程："+Thread.currentThread().getName());  
  
    }  
}  
class ThreadDemo5 implements Callable<String>{  
    @Override  
    public String call() throws Exception {  
        System.out.println("5.实现Callable接口创建线程："+Thread.currentThread().getName());  
        return "Callable线程执行完成";  
    }  
}
