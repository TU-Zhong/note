```java
package NIO;

import java.io.File;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;
import java.nio.ByteBuffer;
import java.nio.channels.FileChannel;
import java.nio.charset.StandardCharsets;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.nio.file.StandardOpenOption;

public class NIO_Test {
    public static void main(String[] args) {
        String sourceFile = "src/resource/source.txt";
        String targetFile = "src/resource/target.txt";
        
        // 确保源文件存在并包含一些测试数据
        createTestFile(sourceFile);
        
        try {
            // 使用Channel读写文件
            channelReadWrite(sourceFile, targetFile);
            
            // 使用Channel复制文件
            channelCopy(sourceFile, "src/resource/copy.txt");
            
            // 使用Files工具类（底层也是NIO）直接操作文件
            directChannelOperation(sourceFile, "src/resource/direct.txt");
            
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    // 创建测试文件
    private static void createTestFile(String filePath) {
        try {
            String content = "Hello NIO Channel!\n这是一个测试文件。\nTest file content.\n";
            Files.write(Paths.get(filePath), 
                       content.getBytes(StandardCharsets.UTF_8), 
                       StandardOpenOption.CREATE, 
                       StandardOpenOption.TRUNCATE_EXISTING);
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    // 使用Channel读写文件
    private static void channelReadWrite(String sourcePath, String targetPath) throws IOException {
        // 创建缓冲区
        ByteBuffer buffer = ByteBuffer.allocate(1024);

        try (FileInputStream fis = new FileInputStream(sourcePath);
             FileOutputStream fos = new FileOutputStream(targetPath);
             // 获取Channel
             FileChannel inChannel = fis.getChannel();
             FileChannel outChannel = fos.getChannel()) {

            while (true) {
                // 清空buffer
                buffer.clear();
                
                // 从Channel中读取数据
                int bytesRead = inChannel.read(buffer);
                if (bytesRead == -1) {
                    break;
                }
                
                // 切换buffer到读模式
                buffer.flip();
                
                // 将buffer中的数据写入Channel
                outChannel.write(buffer);
            }
            
            System.out.println("文件复制完成: " + targetPath);
        }
    }

    // 使用Channel的transferTo方法复制文件
    private static void channelCopy(String sourcePath, String targetPath) throws IOException {
        try (FileChannel sourceChannel = FileChannel.open(Paths.get(sourcePath), StandardOpenOption.READ);
             FileChannel targetChannel = FileChannel.open(Paths.get(targetPath), 
                     StandardOpenOption.CREATE, 
                     StandardOpenOption.WRITE)) {
            
            // 获取文件大小
            long size = sourceChannel.size();
            
            // 使用transferTo直接传输
            sourceChannel.transferTo(0, size, targetChannel);
            
            System.out.println("文件快速复制完成: " + targetPath);
        }
    }

    // 使用Files工具类直接操作文件（底层仍是NIO）
    private static void directChannelOperation(String sourcePath, String targetPath) throws IOException {
        // 读取所有内容
        byte[] content = Files.readAllBytes(Paths.get(sourcePath));
        
        // 写入新文件
        Files.write(Paths.get(targetPath), content, 
                StandardOpenOption.CREATE, 
                StandardOpenOption.TRUNCATE_EXISTING);
        
        System.out.println("直接文件操作完成: " + targetPath);
        
        // 读取文件内容并打印
        System.out.println("\n文件内容：");
        Files.lines(Paths.get(targetPath), StandardCharsets.UTF_8)
             .forEach(System.out::println);
    }
}
```
