	select device_id,gender,age,university, gpa
	from user_profile
	where (university='山东大学' and gpa >3.5)
	    or (university='复旦大学' and gpa>3.8)
	order by device_id;
	//asc,desc 升序，降序排列

	聚合函数 max(属性名)
	[[sql解析]]
	```sql
SELECT 
    COUNT(gender) AS male_num,  -- 统计男性用户数量
    ROUND(AVG(gpa), 1) AS avg_gpa  -- 计算平均 GPA 并四舍五入到 1 位小数
FROM 
    user_profile
WHERE 
    gender = 'male';  -- 筛选出性别为男性的用户
```
