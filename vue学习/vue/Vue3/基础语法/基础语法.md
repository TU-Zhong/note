### 一、基础模板语法
#### 1. 文本插值
```html
<div id="app">
  <p>{{ message }}</p> <!-- 显示数据 -->
  <p>计算属性：{{ count * 2 }}</p> <!-- 支持表达式 -->
</div>
```

#### 2. 原始HTML
```html
<div v-html="rawHtml"></div> <!-- 渲染原始HTML -->
<script>
data() {
  return {
    rawHtml: '<span style="color:red">红色文字</span>'
  }
}
</script>
```

---

### 二、指令（Directives）
#### 1. 条件渲染
```html
<div v-if="score > 90">优秀</div>
<div v-else-if="score > 60">及格</div>
<div v-else>不及格</div>

<!-- v-show 通过CSS控制显示 -->
<p v-show="isVisible">你看得见我</p>
```

#### 2. 列表渲染
```html
<ul>
  <!-- 遍历数组 -->
  <li v-for="(item, index) in items" :key="item.id">
    {{ index }}. {{ item.name }}
  </li>
  
  <!-- 遍历对象 -->
  <li v-for="(value, key) in userInfo">
    {{ key }}: {{ value }}
  </li>
</ul>
```

### 三、事件处理
#### 1. 基本用法
```html
<button @click="count += 1">点击+1</button>

<!-- 调用方法 -->
<button @click="sayHello('Vue3')">打招呼</button>

<script>
methods: {
  sayHello(name) {
    alert(`Hello ${name}!`)
  }
}
</script>
```

#### 2. 事件修饰符
```html
<!-- 阻止默认行为 -->
<form @submit.prevent="onSubmit"></form>

<!-- 按键修饰符 -->
<input @keyup.enter="submit">
```

---

### 四、表单输入绑定
#### 1. 基础双向绑定
```html
<input v-model="text" placeholder="输入内容">
<p>你输入了：{{ text }}</p>

<script>
data() {
  return {
    text: ''
  }
}
</script>
```

#### 2. 复选框
```html
<input type="checkbox" v-model="checked">
<label>{{ checked ? '已选' : '未选' }}</label>
```

#### 3. 单选/多选
```html
<!-- 单选 -->
<select v-model="selected">
  <option>A</option>
  <option>B</option>
</select>

<!-- 多选 -->
<select v-model="selectedList" multiple>
  <option>苹果</option>
  <option>香蕉</option>
</select>
```

### 五、属性绑定
#### 1. 动态绑定属性
```html
<a :href="url">链接</a>
<img :src="imageSrc">
<div :class="{ active: isActive }"></div>
```

#### 2. 样式绑定
```html
<!-- 对象语法 -->
<div :style="{ color: textColor, fontSize: size + 'px' }"></div>

<!-- 数组语法 -->
<div :class="[errorClass, { 'text-bold': isBold }]"></div>
```

### 六、计算属性
```html
<p>总价：{{ totalPrice }}</p>

<script>
computed: {
  totalPrice() {
    return this.price * this.quantity
  }
}
</script>
```

### 七、生命周期（常用）
```javascript
export default {
  created() {
    console.log('组件已创建，但未挂载到DOM')
  },
  mounted() {
    console.log('组件已挂载到DOM')
  },
  updated() {
    console.log('数据更新导致DOM变化')
  },
  unmounted() {
    console.log('组件已卸载')
  }
}
```

### 八、组件基础
#### 1. 注册组件
```javascript
// 全局组件
const app = createApp({})
app.component('my-button', {
  template: `<button @click="$emit('btn-click')">点击</button>`
})
```

#### 2. Props 传递
```html
<!-- 父组件 -->
<child-component :title="parentTitle"></child-component>

<!-- 子组件 -->
<script>
props: ['title'],
template: `<h2>{{ title }}</h2>`
</script>
```

### 九、综合示例：计数器
```html
<div id="app">
  <button @click="decrement">-</button>
  <span>{{ count }}</span>
  <button @click="increment">+</button>
</div>

<script>
const { createApp } = Vue

createApp({
  data() {
    return { count: 0 }
  },
  methods: {
    increment() {
      this.count++
    },
    decrement() {
      this.count--
    }
  }
}).mount('#app')
</script>
```

### 📚 基础语法总结表
| 分类          | 核心语法                      | 作用                          |
|---------------|-----------------------------|-----------------------------|
| 数据绑定       | `{{ }}` / `v-text`          | 显示文本内容                    |
| 属性绑定       | `v-bind` 或 `:`             | 动态绑定HTML属性                |
| 事件绑定       | `v-on` 或 `@`               | 监听DOM事件                   |
| 双向绑定       | `v-model`                   | 表单输入与应用状态同步            |
| 条件渲染       | `v-if` / `v-else` / `v-show` | 条件显示元素                   |
| 列表渲染       | `v-for`                     | 遍历数组/对象生成元素            |
| 计算属性       | `computed`                  | 基于响应式数据的复杂计算          |
