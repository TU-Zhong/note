<template>
  <div class="db-sidebar">
    <div class="sidebar-toolbar">
      <el-tooltip content="新建查询控制台" placement="bottom">
        <el-icon class="action-icon" @click="handleNewConsole"><Plus /></el-icon>

<script setup>
import { ref, onMounted, watch } from 'vue'

import { useWorkspaceStore } from '../../stores/useWorkspaceStore'
// 注意：这里假设你已经封装了 axios 请求，如果没有，可以先用 mock 数据
// import { getMetadataTree } from '../../api/metadata' 

const store = useWorkspaceStore()
const filterText = ref('')
const treeRef = ref()
const treeData = ref([])

const defaultProps = {
  children: 'children',
  label: 'label',
}

// --- 逻辑 1：新建控制台 ---
const handleNewConsole = () => {
  store.addTab({
    title: 'console-1',
    type: 'console',
    params: { sql: '' }
  })
}

// --- 逻辑 2：获取后端数据 ---
const fetchTreeData = async () => {
  // 模拟从后端接口获取数据
  // 实际开发时替换为：treeData.value = await getMetadataTree()
  treeData.value = [
    {
      id: 'db_1',
      label: 'mysql_production',
      type: 'database',
      children: [
        { id: 't_1', label: 't_student', type: 'table' },
        { id: 't_2', label: 't_course', type: 'table' }
      ]
    }
  ]
}

// --- 逻辑 3：双击打开表数据 ---
const handleNodeDblClick = (data) => {
  if (data.type === 'table') {
    store.addTab({
      id: data.id,
      title: data.label,
      type: 'table',
      params: { tableName: data.label }
    })
  }
}

// --- 逻辑 4：搜索过滤 ---
watch(filterText, (val) => {
  treeRef.value?.filter(val)
})
const filterNode = (value, data) => {
  if (!value) return true
  return data.label.includes(value)
}

onMounted(() => {
  fetchTreeData()
})
</script>

### 🔍 关键细节详解（边写边理解）

#### 1. 为什么用 `node-dblclick` 而不是 `click`？

在专业的 IDE 中，**单击**通常只是“选中”（为了在下方显示属性），**双击**才是“打开”。这能防止用户只是想点开折叠箭头时，不小心触发了右侧打开大量标签页的 Bug。

#### 2. 状态驱动的图标

代码中使用了 `v-if="data.type === 'database'"`。在你的毕业设计中，后端返回的树形结构通常包含不同的层级（库 -> 表 -> 列）。通过 `type` 来渲染不同的图标（Coin 或 Grid），会让你的软件看起来非常精致，不仅是“能用”，而是“好用”。

#### 3. 与 Pinia 的“化学反应”

观察 `handleNewConsole` 函数。它并没有直接去操作 DOM，只是往 `store.addTab` 里丢了一个对象。

- **这就是解耦**：Sidebar 只管“发号施令”，右侧的 `WorkArea` 会自动感应到并画出 Tab。
