```js
import { defineStore } from 'pinia'
import { ref } from 'vue'

export const useWorkspaceStore = defineStore('workspace', () => {
  // --- 2. 定义核心状态 (State) ---
  
  // 存储所有已打开标签的数组
  // 每个对象格式：{ id: 'unique_id', title: '表名/Console', type: 'table/console', params: {} }
  const tabs = ref([])
  
  // 记录当前选中的标签 ID
  const activeTabId = ref('')

  // --- 3. 编写核心动作 (Actions) ---

  /**
   * 添加一个新的标签页 (IDEA 逻辑：判断是否存在 -> 跳转或新增)
   * @param {Object} node 传入的节点信息，包含 title, type, 以及业务参数
   */
  const addTab = (node) => {
    // 1. 生成或获取唯一 ID (如果是表，直接用表名作为 ID 防止重复打开同一个表)
    const tabId = node.id || `${node.type}_${node.title}`

    // 2. 判断该 Tab 是否已经存在
    const existingTab = tabs.value.find(tab => tab.id === tabId)

    if (existingTab) {
      // 若已存在，直接激活它
      activeTabId.value = tabId
    } else {
      // 若不存在，则 push 到数组并激活
      tabs.value.push({
        id: tabId,
        title: node.title,
        type: node.type,
        params: node.params || {}
      })
      activeTabId.value = tabId
    }
  }

  /**
   * 关闭指定的标签页 (IDEA 逻辑：智能跳转激活项)
   * @param {String} targetId 要关闭的 Tab ID
   */
  const removeTab = (targetId) => {
    const tabList = tabs.value
    let activeId = activeTabId.value

    // 1. 如果关闭的是当前激活的 Tab，需要计算“接班人”
    if (activeId === targetId) {
      tabList.forEach((tab, index) => {
        if (tab.id === targetId) {
          // 尝试寻找下一个标签，若没下一个就找上一个
          const nextTab = tabList[index + 1] || tabList[index - 1]
          if (nextTab) {
            activeId = nextTab.id
          } else {
            activeId = '' // 没标签了
          }
        }
      })
    }

    // 2. 更新激活状态并剔除数组
    activeTabId.value = activeId
    tabs.value = tabList.filter(tab => tab.id !== targetId)
  }

  return {
    tabs,
    activeTabId,
    addTab,
    removeTab
  }
})
```

### 💡 核心逻辑说明（用于论文编写）

1. **唯一 ID 策略**：在 `addTab` 中，我们并没有简单使用 `Date.now()`。对于“表数据查看”类型的 Tab，建议使用 `表名` 作为 ID。这样当用户反复点击同一张表时，系统会**高亮已有的 Tab** 而不是开启无数个重复页面，这完全符合 IDEA 的操作规范。
    
2. **“接班人”算法**：在 `removeTab` 中，如果关闭当前页，逻辑会优先寻找右侧相邻项，若右侧为空则寻找左侧。这种**平滑切换**保证了用户在管理多个 SQL 控制台时不会因为关闭操作而突然“断片”。
