<script setup>  
import { ref } from 'vue'  
import {  
  Plus, Setting, Refresh, CircleClose, VideoPlay,  
  Grid, Connection, Coin, Memo  
} from '@element-plus/icons-vue'  
  
// --- 树形数据逻辑 ---const dbTreeData = ref([  
  {    label: '@localhost',  
    level: 'server',  
    countInfo: '2 of 5',  
    children: [  
      { label: 'master', level: 'database', countInfo: '1 of 13' },  
      {  
        label: 'School_DB',  
        level: 'database',  
        countInfo: '1 of 13',  
        children: [  
          { label: 'Student', level: 'table' },  
          { label: 'Course', level: 'table' }  
        ]      }    ]  }])  
const treeProps = { children: 'children', label: 'label' }  
  
// --- 标签页 (Tabs) 逻辑 ---const activeTab = ref('')  
const tabs = ref([])  
  
const addTab = (type) => {  
  const newName = `${type}_${Date.now()}`  
  tabs.value.push({  
    title: type === 'console' ? `console_${tabs.value.length + 1}` : 'DDL Viewer',  
    name: newName,  
    type: type  
  })  
  activeTab.value = newName  
}  
  
const removeTab = (targetName) => {  
  tabs.value = tabs.value.filter(tab => tab.name !== targetName)  
  if (activeTab.value === targetName && tabs.value.length > 0) {  
    activeTab.value = tabs.value[tabs.value.length - 1].name  
  }  
}  
  
const handleNodeClick = (data) => {  
  if (data.level === 'table') {  
    // 双击或点击表名时增加一个数据查看 Tab    const exist = tabs.value.find(t => t.title === data.label)  
    if (!exist) {  
      tabs.value.push({ title: data.label, name: data.label, type: 'table' })  
    }  
    activeTab.value = data.label  
  }  
}  
</script>
