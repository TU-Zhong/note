```java
package com.example.dbdesign.service;

import org.springframework.stereotype.Service;

import java.io.File;
import java.net.URL;
import java.net.URLClassLoader;
import java.sql.Connection;
import java.sql.Driver;
import java.util.Properties;

@Service
public class DriverLoaderService {

    /**
     * 核心黑科技：不通过 DriverManager，而是直接用类加载器加载驱动并连接
     *
     * @param jarPath  驱动 Jar 包的绝对路径 (Day 3 上传得到的那个路径)
     * @param url      JDBC 连接字符串
     * @param username 数据库账号
     * @param password 数据库密码
     * @return 数据库连接对象 (Connection)
     */
    public Connection getDynamicConnection(String jarPath, String url, String username, String password) throws Exception {
        // 1. 校验文件是否存在
        File jarFile = new File(jarPath);
        if (!jarFile.exists()) {
            throw new RuntimeException("驱动文件不存在: " + jarPath);
        }

        // 2. 【核心技术点】创建隔离的类加载器
        // 我们用 URLClassLoader 加载外部 Jar，这样 Spring Boot 根本不需要知道这个 Jar 的存在
        URL[] urls = new URL[]{jarFile.toURI().toURL()};
        URLClassLoader ucl = new URLClassLoader(urls, Thread.currentThread().getContextClassLoader());
```

```java


        // 3. 【核心技术点】反射加载驱动类
        // SQL Server 的驱动类名是固定的，如果是其他数据库，这个类名也可以让前端传
        //微软的设计哲学是“向后兼容”。这意味着你用代码加载这个类时，不需要关心对方是哪个版本的数据库。
        String driverClassName = "com.microsoft.sqlserver.jdbc.SQLServerDriver";
        Class<?> driverClass = ucl.loadClass(driverClassName);

        // 4. 实例化驱动对象
        // 相当于 new SQLServerDriver()，但我们是在运行时动态 new 的
        Driver driver = (Driver) driverClass.getDeclaredConstructor().newInstance();

        // 5. 【核心技术点】绕过 DriverManager 直接连接
        // 因为 DriverManager 只能管理启动时加载的驱动，我们要手动调用 driver.connect
        Properties props = new Properties();
        props.setProperty("user", username);
        props.setProperty("password", password);
        
        // 针对 SQL Server Docker 版的特殊配置（避免 SSL 报错）
        if (!url.contains("encrypt=")) {
            props.setProperty("encrypt", "false");
        }
        if (!url.contains("trustServerCertificate=")) {
            props.setProperty("trustServerCertificate", "true");
        }

        Connection connection = driver.connect(url, props);
        return connection;
    }
}
```

#### 2. 升级控制器 (`DriverController.java`)

我们需要在之前的 `DriverController` 里加一个接口，用来测试这个 Service 是否好用。

打开 `DriverController.java`，**修改**成下面这样（添加了 Service 注入和新的测试接口）：

Java

```java
package com.example.dbdesign.controller;

import com.example.dbdesign.service.DriverLoaderService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;
import org.springframework.web.multipart.MultipartFile;

import java.io.File;
import java.io.IOException;
import java.sql.Connection;

@RestController
@RequestMapping("/api/driver")
@CrossOrigin
public class DriverController {

    private static final String UPLOAD_DIR = System.getProperty("user.dir") + File.separator + "drivers";

    // 注入刚才写的服务
    @Autowired
    private DriverLoaderService driverLoaderService;

    // ... (之前的 uploadDriver 方法保留不变) ...
    @PostMapping("/upload")
    public String uploadDriver(@RequestParam("file") MultipartFile file) {
        // ... (保持你之前的代码不变) ...
        if (file.isEmpty()) return "文件为空";
        try {
            File dir = new File(UPLOAD_DIR);
            if (!dir.exists()) dir.mkdirs();
            File dest = new File(UPLOAD_DIR + File.separator + file.getOriginalFilename());
            file.transferTo(dest);
            return dest.getAbsolutePath(); // 返回绝对路径
        } catch (IOException e) {
            return "Error: " + e.getMessage();
        }
    }

    /**
     * 新增：测试连接接口
     */
    @PostMapping("/connect")
    public String testConnection(
            @RequestParam("jarPath") String jarPath,
            @RequestParam("url") String url,
            @RequestParam("user") String user,
            @RequestParam("password") String password) {
        
        try (Connection conn = driverLoaderService.getDynamicConnection(jarPath, url, user, password)) {
            // 如果能运行到这里，说明连接成功！
            return "✅ 连接成功！数据库名称: " + conn.getMetaData().getDatabaseProductName() + 
                   ", 版本: " + conn.getMetaData().getDatabaseProductVersion();
        } catch (Exception e) {
            e.printStackTrace(); // 看控制台报错详情
            return "❌ 连接失败: " + e.getMessage();
        }
    }
}
```
