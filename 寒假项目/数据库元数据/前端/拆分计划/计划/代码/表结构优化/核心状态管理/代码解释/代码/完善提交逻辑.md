#### 2. 在 `<script setup>` 中引入图标并添加逻辑
添加以下代码来支撑工具栏的交互。你需要引入 `ElMessageBox` 来做刷新时的安全警告：

```javascript
// 别忘了在最上面引入图标和弹窗组件
import { Plus, Minus, Refresh, Check, Grid } from '@element-plus/icons-vue'
import { ElMessage, ElMessageBox } from 'element-plus'

// ... 前面定义的 tableData, diffPool, fetchData 等保持不变 ...

// === 工具栏与交互逻辑 ===

// 1. 计算属性：实时计算是否有未提交的更改
const hasChanges = computed(() => {
  return diffPool.inserted.size > 0 || diffPool.updated.size > 0 || diffPool.deleted.size > 0
})
// 计算总共修改了多少行（用于按钮上显示数字）
const totalChanges = computed(() => {
  return diffPool.inserted.size + diffPool.updated.size + diffPool.deleted.size
})

// 2. 记录当前鼠标单击选中的是哪一行
const currentRow = ref(null)
const handleCurrentRowChange = (val) => {
  currentRow.value = val
}

// 计算当前是否可以点击删除按钮
const canDelete = computed(() => {
  // 必须选中了一行，且这行还没被放入删除池
  return currentRow.value && !diffPool.deleted.has(currentRow.value.__row_id)
})

// 3. 真实的添加行逻辑
const handleAddRow = () => {
  const newRow = { __row_id: generateRowId() }
  tableColumns.value.forEach(col => {
    newRow[col] = null
  })
  tableData.value.unshift(newRow) // 插入到表格最上面
  diffPool.inserted.add(newRow.__row_id)
}

// 4. 真实的删除行逻辑
const handleDeleteRow = () => {
  if (!currentRow.value) return
  const rowId = currentRow.value.__row_id

  if (diffPool.inserted.has(rowId)) {
    // 如果是刚新增的虚空行，直接从数组抹除
    const index = tableData.value.findIndex(r => r.__row_id === rowId)
    if (index !== -1) tableData.value.splice(index, 1)
    diffPool.inserted.delete(rowId)
  } else {
    // 如果是老数据，放入删除池（变红）
    diffPool.deleted.add(rowId)
  }
  // 删完后取消高亮选中
  currentRow.value = null
}

// 5. 刷新逻辑（带安全警告）
const handleRefresh = () => {
  if (hasChanges.value) {
    ElMessageBox.confirm(
      '您有未提交的更改，刷新将丢失所有修改。是否继续刷新？',
      '警告',
      { confirmButtonText: '确定丢失', cancelButtonText: '取消', type: 'warning' }
    ).then(() => {
      fetchData() // 用户确认，重新拉取数据（fetchData内部会清空 diffPool）
    }).catch(() => {
      // 用户取消，无事发生
    })
  } else {
    // 没有更改，直接丝滑刷新
    fetchData()
  }
}

// 6. 提交逻辑的空壳（下一阶段联调后端用的）
const handleSubmit = () => {
  console.log("准备提交的数据：", {
    inserted: Array.from(diffPool.inserted),
    updated: Array.from(diffPool.updated),
    deleted: Array.from(diffPool.deleted)
  })
  ElMessage.success("数据已准备好，等待对接后端！")
}
````
