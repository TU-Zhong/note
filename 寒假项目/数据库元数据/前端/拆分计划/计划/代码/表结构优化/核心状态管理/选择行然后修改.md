### 🛠️ 核心改造：实现“双击单元格变输入框”

```js
// === 真实的表格内联编辑逻辑 ===

// 记录当前正在编辑的单元格坐标
const editingCell = ref({
  rowId: null,
  colKey: null
})

// 表格自带的事件：双击某个单元格时触发
const handleCellDblClick = (row, column, cell, event) => {
  // 如果这行已经被删除了，不允许编辑
  if (diffPool.deleted.has(row.__row_id)) return
  
  // 记录下当前双击的是哪一行的哪一列
  editingCell.value = {
    rowId: row.__row_id,
    colKey: column.property
  }
}

// 输入框失去焦点，或者按下回车键时触发
const handleInputBlur = (row) => {
  // 1. 关闭输入框模式，变回普通文本
  editingCell.value = { rowId: null, colKey: null }
  
  // 2. 核心状态判定：这行到底有没有被修改？
  if (!diffPool.inserted.has(row.__row_id)) {
    // 只有当它不是新插入的空行时，才考虑把它加入 updated 修改池
    
    // 【进阶优化】这里其实应该拿 row 的新值，去和 originalData (源数据快照) 里对应的那行对比
    // 如果发现值真的变了，才打上 update 标签。为了让你先看效果，只要编辑了我们就当它改了。
    diffPool.updated.add(row.__row_id)
  }
}
```
