<template>
  <div class="data-viewer" v-loading="loading">
    <div class="toolbar">
      <el-button type="primary" link @click="fetchData">
        <el-icon><Refresh /></el-icon> 刷新数据
      </el-button>
      <el-divider direction="vertical" />
      <span class="info">
        当前表: 
        <span class="table-name">
          {{ params.schema ? `${params.schema}.` : '' }}{{ params.tableName }}
        </span>
      </span>
      <span class="count" v-if="tableData.length > 0">
        (前 {{ tableData.length }} 条预览)
      </span>
    </div>

    <div class="table-container">
      <el-table 
        :data="tableData" 
        style="width: 100%; height: 100%" 
        border 
        stripe
        highlight-current-row
      >
        <el-table-column
          v-for="col in tableColumns"
          :key="col"
          :prop="col"
          :label="col"
          min-width="120"
          show-overflow-tooltip
          sortable
        />

        <template #empty>
          <el-empty :image-size="100" description="暂无数据" />
        </template>
      </el-table>
    </div>
  </div>
</template>

<script setup>
import { ref, onMounted } from 'vue'
import axios from 'axios'
import { ElMessage } from 'element-plus'
import { Refresh } from '@element-plus/icons-vue' // 记得引入图标

// 接收父组件(Tab)传来的参数
const props = defineProps({
  tabId: String,
  params: {
    type: Object,
    default: () => ({ tableName: '', schema: '' })
  }
})

// 定义响应式数据
const tableColumns = ref([]) // 存放列名 ['id', 'name', ...]
const tableData = ref([])    // 存放行数据 [{id:1, name:'...'}, ...]
const loading = ref(false)   // 加载状态

// 核心获取数据方法
const fetchData = async () => {
  // 安全校验
  if (!props.params.tableName) {
    ElMessage.warning('未指定表名')
    return
  }
  
  loading.value = true
  
  try {
    // 发起 GET 请求
    // 假设你的后端运行在 8080 端口
    const res = await axios.get('http://localhost:8080/api/metadata/table/data', {
      headers: {
        // 从 localStorage 获取 Token (根据你登录逻辑实际存储的位置修改)
        'Authorization': localStorage.getItem('token') || '' 
      },
      params: {
        // @RequestParam 接收的参数
        tableName: props.params.tableName,
        schema: props.params.schema || '' // 没传schema就给空字符串
      }
    })
    
    // 处理响应
    if (res.data.code === 200 || res.data.code === 1) { // 兼容不同的成功状态码
      const resultData = res.data.data
      
      // 1. 设置表头
      if (resultData.columns && Array.isArray(resultData.columns)) {
        tableColumns.value = resultData.columns
      }
      
      // 2. 设置数据行
      if (resultData.rows && Array.isArray(resultData.rows)) {
        tableData.value = resultData.rows
      }
      
      ElMessage.success('查询成功')
    } else {
      // 业务错误 (如表不存在)
      ElMessage.error(res.data.msg || '数据加载失败')
    }
  } catch (error) {
    console.error('Fetch Error:', error)
    // 提取错误信息
    const errorMsg = error.response?.data?.msg || error.message || '网络请求异常'
    ElMessage.error(errorMsg)
  } finally {
    loading.value = false
  }
}

// 组件挂载完成后自动调用
onMounted(() => {
  fetchData()
})
</script>

  padding: 10px;

### 💡 几个需要注意的细节：

1. **Token 获取**：代码里写的是 `localStorage.getItem('token')`。请确认你的登录页面是把 Token 存在这里的，如果不是（比如叫 `satoken` 或 `Authorization`），请自行修改这一行。
    
2. **图标引入**：我引入了 Element Plus 的 `Refresh` 图标。如果你的项目报错找不到图标，可能需要检查是否全局注册了，或者按需引入一下。
    
3. **样式布局**：使用了 `flex: 1` 配合 `height: 100%`，确保当数据很多时，表格会在容器内部滚动，而不是把整个页面撑长。
