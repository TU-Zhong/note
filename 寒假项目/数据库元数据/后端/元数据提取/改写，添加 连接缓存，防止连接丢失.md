### 为什么 Controller 要改？

你现在的 Controller 写法是 “连上 -> 打印一句话 -> 自动挂断”（因为你用了 try(Connection ...)）。

为了做 Day 7 的“获取表结构”，你必须让连接 “保持通话”，并把这个通话的“号码牌”（Token）给前端。

你需要做三件事：

1. **新建** `ConnectionCache` 类（用来存连接）。
    
2. **修改** `DriverController`（注入缓存，返回 Token）。
    
3. **微调** Service 里的变量名（强迫症建议，不改也能跑）。

### 第一步：新建 ConnectionCache.java

在 `Service` 包下新建这个类。这是你的“连接仓库”。

Java

```java
package com.example.dbdesign.Service;

import org.springframework.stereotype.Component;
import java.sql.Connection;
import java.sql.SQLException;
import java.util.Map;
import java.util.UUID;
import java.util.concurrent.ConcurrentHashMap;

@Component
public class ConnectionCache {
    // 线程安全的 Map，用来存活着的连接
    private final Map<String, Connection> connectionMap = new ConcurrentHashMap<>();

    // 存连接，返回 ID
    public String addConnection(Connection connection) {
        String id = UUID.randomUUID().toString();
        connectionMap.put(id, connection);
        return id;
    }

    // 取连接
    public Connection getConnection(String id) {
        return connectionMap.get(id);
    }

    // 删连接
    public void removeConnection(String id) {
        Connection conn = connectionMap.remove(id);
        if (conn != null) {
            try { conn.close(); } catch (SQLException e) { e.printStackTrace(); }
        }
    }
}
```

### 第二步：修改 DriverController.java (重头戏)

请直接替换你的 Controller 代码。注意看我改了哪里：

```java
package com.example.dbdesign.Controller;

import com.example.dbdesign.Service.ConnectionCache; // 导入缓存类
import com.example.dbdesign.Service.DriverLoaderService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;
import org.springframework.web.multipart.MultipartFile;

import java.io.File;
import java.io.IOException;
import java.sql.Connection;
import java.util.HashMap;
import java.util.Map;

@RestController
@RequestMapping("/api/driver")
@CrossOrigin
public class DriverController {
    private static final String UPLOAD_DIR = System.getProperty("user.dir") + File.separator + "drivers";

    @Autowired
    private DriverLoaderService driverLoaderService;

    // ✅ 新增：注入连接缓存
    @Autowired
    private ConnectionCache connectionCache;

    @PostMapping("/upload")
    public String uploadDriver(@RequestParam("file") MultipartFile file){
        // ... 上传逻辑不用变，你写的没问题 ...
        if(file.isEmpty()){
            return "上传失败，请选择文件";
        }
        try{
            File directory=new File(UPLOAD_DIR);
            if(!directory.exists()){
                directory.mkdirs(); // 建议用 mkdirs() 防止父目录不存在
            }
            String originFileName=file.getOriginalFilename();
            File destFile=new File(UPLOAD_DIR+ File.separator+originFileName);
            file.transferTo(destFile);
            return destFile.getAbsolutePath();
        }catch (IOException e){
            e.printStackTrace();
            return "上传出错"+e.getMessage();
        }
    }

    // 🚨 这里的返回值改成了 Map，方便前端获取 Token
    @PostMapping("/connect")
    public Map<String, Object> connectDatabase(
            @RequestParam("jarPath") String jarPath,
            @RequestParam("url") String url,
            @RequestParam("userName") String userName,
            @RequestParam("passWord") String passWord // ❌ 修正了你原来的 "passWord]" 拼写错误
    ){
        Map<String, Object> result = new HashMap<>();

        try {
            // 1. 获取连接 (注意：这里【不要】用 try(...)，因为我们要让连接活着)
            Connection conn = driverLoaderService.getDynamicConnection(jarPath, url, userName, passWord);

            // 2. 将活着的连接存入缓存，拿到唯一的 Token
            String token = connectionCache.addConnection(conn);

            // 3. 返回成功信息和 Token
            result.put("success", true);
            result.put("msg", "连接成功");
            result.put("token", token); // 🔑 关键：前端下次要带着这个来
            result.put("dbType", conn.getMetaData().getDatabaseProductName());
            result.put("dbVersion", conn.getMetaData().getDatabaseProductVersion());

        } catch (Exception e) {
            e.printStackTrace();
            result.put("success", false);
            result.put("msg", "连接失败: " + e.getMessage());
        }

        return result;
    }
}
```
