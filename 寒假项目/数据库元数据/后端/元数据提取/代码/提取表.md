```java
package com.example.dbdesign.service;

import com.example.dbdesign.vo.TableVO;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.sql.Connection;
import java.sql.DatabaseMetaData;
import java.sql.ResultSet;
import java.util.ArrayList;
import java.util.List;

@Service
public class MetadataService {

    @Autowired
    private ConnectionCache connectionCache;

    /**
     * 获取指定连接下的所有表和视图
     * @param token 前端传来的连接令牌
     * @param schemaFilter 模式过滤 (如 "dbo")，传 null 则查询所有
     */
    public List<TableVO> getTables(String token, String schemaFilter) throws Exception {
        // 1. 从缓存获取连接
        Connection conn = connectionCache.getConnection(token);
        if (conn == null) {
            throw new RuntimeException("连接已失效或不存在，请重新连接");
        }

        // 2. 获取元数据对象
        DatabaseMetaData metaData = conn.getMetaData();

        // 3. 准备参数
        // Catalog: 通常传 null (对于 SQL Server 是数据库名，但连接时已指定，所以 null 即可)
        // Schema: 模式名 (如 "dbo")，如果前端没传，可以用 null 查所有，或者默认 "dbo"
        // TableNamePattern: "%" 表示匹配所有表
        // Types: 只查表和视图，过滤掉系统表
        String[] types = {"TABLE", "VIEW"};
        
        // 4. 执行查询
        ResultSet rs = metaData.getTables(null, schemaFilter, "%", types);

        // 5. 封装结果
        List<TableVO> tableList = new ArrayList<>();
        while (rs.next()) {
            String tableName = rs.getString("TABLE_NAME");
            String tableType = rs.getString("TABLE_TYPE");
            String remarks = rs.getString("REMARKS");
            String tableSchema = rs.getString("TABLE_SCHEM");

            TableVO vo = new TableVO(tableName, tableType, remarks, tableSchema);
            tableList.add(vo);
        }
        
        // 6. 记得关闭 ResultSet (Connection 不能关，因为要留着下次用)
        rs.close();

        return tableList;
    }
}
```

### 第三步：编写接口 (Controller)

在 `com.example.dbdesign.controller` 包下新建 `MetadataController.java`。

**注意**：

1. 我们要从请求头 `Authorization` 或参数里拿到 `token`。
    
2. 允许前端传 `schema`，如果不传，我们在代码里决定是给 `null` 还是默认值。
    

Java

```java
package com.example.dbdesign.controller;

import com.example.dbdesign.common.Result;
import com.example.dbdesign.service.MetadataService;
import com.example.dbdesign.vo.TableVO;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;

import java.util.List;

@RestController
@RequestMapping("/api/metadata")
@CrossOrigin
public class MetadataController {

    @Autowired
    private MetadataService metadataService;

    /**
     * 获取表列表
     * @param token 在 Header 中传递 (Authorization: uuid-xxxx) 或者作为参数传递都可以
     * @param schema (可选) 如果不传，SQL Server 默认可能会查出很多 schema，建议前端传 "dbo"
     */
    @GetMapping("/tables")
    public Result<List<TableVO>> getTables(
            @RequestHeader(value = "Authorization", required = false) String token, 
            @RequestParam(required = false) String schema
    ) {
        // 兼容性处理：如果 Header 没传，也可以尝试从参数拿 (看你前端怎么写方便)
        if (token == null || token.isEmpty()) {
            return Result.error("未授权：请提供连接 Token");
        }

        // 处理 Schema 默认值
        // 对于 SQL Server，如果不传 schema，传 null 会查出 guest, sys 等用户的表
        // 建议：如果没传，默认设为 "dbo" (这是 SQL Server 的默认用户模式)
        if (schema == null || schema.trim().isEmpty()) {
            schema = "dbo"; 
        }

        try {
            List<TableVO> tables = metadataService.getTables(token, schema);
            return Result.success("获取表结构成功", tables);
        } catch (Exception e) {
            e.printStackTrace();
            return Result.error("获取失败: " + e.getMessage());
        }
    }
}
```

1. 先登录拿到 Token

先调用 /api/driver/connect 接口，复制返回结果里的 token。

**2. 调用获取表接口**

- **Method**: `GET`
    
- **URL**: `http://localhost:8080/api/metadata/tables`
    
- **Headers**:
    
    - Key: `Authorization`
        
    - Value: `粘贴刚才的Token`
        
- **Params (Query Params)**:
    
    - Key: `schema`
        
    - Value: `dbo` (你可以试着不传这个参数，看看会不会多出很多奇怪的表)
