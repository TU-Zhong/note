# 合并（仅当列结构一致）
if len(dfs) == 1:
    df = list(dfs.values())[0]

    cols_sets = [tuple(map(str.lower, d.columns)) for d in dfs.values()]

    df = pd.concat(dfs.values(), ignore_index=True) if same_structure else max(dfs.values(), key=lambda d: d.shape[1])

df.columns = [str(c).strip().replace("\n"," ").replace("\r"," ") for c in df.columns]
