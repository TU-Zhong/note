import java.util.HashSet;
import java.util.Set;

/**
 * LeetCode-like problem solution.
 */
class Solution {

    /**
     * 判断这排文物的所属朝代编号是否能够视为连续的五个朝代。
     *
     * @param places 包含5个文物所属朝代编号的数组。
     * @return 如果可以视为连续，返回 true；否则返回 false。
     */
    public boolean checkDynasty(int[] places) {
        // 题目规定数组长度为5，但增加一个健壮性检查没有坏处。
        if (places == null || places.length != 5) {
            return false;
        }

        // 使用 Set 来快速检测是否存在重复的朝代编号。
        Set<Integer> seen = new HashSet<>();
        
        // 初始化最大值和最小值。
        // 由于朝代编号从1开始，可以将max初始化为0或-1，min初始化为一个较大的数。
        int maxVal = 0;
        int minVal = 14; // 根据题意，朝代编号为1-13，14肯定比所有有效编号都大。

        for (int dynastyId : places) {
            // 0 是万能牌，跳过它。
            if (dynastyId == 0) {
                continue;
            }

            // 如果在 Set 中已存在该编号，说明有重复，不可能连续，直接返回 false。
            if (seen.contains(dynastyId)) {
                return false;
            }
            
            // 将当前编号加入 Set。
            seen.add(dynastyId);

            // 更新遇到的最大和最小朝代编号。
            maxVal = Math.max(maxVal, dynastyId);
            minVal = Math.min(minVal, dynastyId);
        }

        // 如果 seen 为空，说明数组中全是 0，可以构成任意连续序列，返回 true。
        if (seen.isEmpty()) {
            return true;
        }

        // 核心判断：最大值与最小值的差值如果小于5，则可以构成连续序列。
        return maxVal - minVal < 5;
    }
}

// --- 主函数用于测试 ---
public class Main {
    public static void main(String[] args) {
        Solution solution = new Solution();

        // 示例 1：
        int[] places1 = {0, 6, 9, 0, 7};
        // 排序后非0部分为 [6, 7, 9]。max=9, min=6。max-min = 3 < 5。可以构成 5,6,7,8,9 或 6,7,8,9,10。
        System.out.println("示例 1 [0, 6, 9, 0, 7] -> " + solution.checkDynasty(places1)); // 预期输出: True

        // 示例 2：
        int[] places2 = {7, 8, 9, 10, 11};
        // max=11, min=7。max-min = 4 < 5。
        System.out.println("示例 2 [7, 8, 9, 10, 11] -> " + solution.checkDynasty(places2)); // 预期输出: True

        // 其他测试用例
        int[] places3 = {7, 7, 8, 0, 9};
        // 存在重复的 7，应返回 false。
        System.out.println("测试用例 [7, 7, 8, 0, 9] -> " + solution.checkDynasty(places3)); // 预期输出: False

        int[] places4 = {1, 2, 3, 4, 6};
        // max=6, min=1。max-min = 5，差值不小于5，无法用一个0填补。
        System.out.println("测试用例 [1, 2, 3, 4, 6] -> " + solution.checkDynasty(places4)); // 预期输出: False
        
        int[] places5 = {0, 0, 0, 0, 0};
        // 全部是万能牌，当然可以。
        System.out.println("测试用例 [0, 0, 0, 0, 0] -> " + solution.checkDynasty(places5)); // 预期输出: True
    }
}
