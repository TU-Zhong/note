在 MySQL 的执行顺序中 WHERE 早于 SELECT，不能在同层 WHERE 中引用该别名，因此语句无效（逻辑也不成立）。

```java
public class Solution {
    
    public long countMultiples(long N, long a, long b) {
        long countA = N / a;
        long countB = N / b;
        
        // 计算 a 和 b 的最小公倍数 (LCM)
        // 注意：写法是 a / gcd(a,b) * b，先除后乘可以有效防止 long 溢出
        long lcm = (a / gcd(a, b)) * b;
        long countCommon = N / lcm;
        
        // 容斥原理：总数 = A + B - 交集
        return countA + countB - countCommon;
    }

    // 经典的辗转相除法求最大公约数 (GCD)
    private long gcd(long a, long b) {
        return b == 0 ? a : gcd(b, a % b);
    }
}
```
